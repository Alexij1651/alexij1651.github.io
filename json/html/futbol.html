// Función que recorre el array y construye la tabla en el DOM
function generarTabla(datos) {
 
  // 1. Seleccionamos el contenedor donde inyectaremos la tabla
  const contenedor = document.getElementById("tabla-container");
 
  // 2. Construimos la cabecera con los nombres de los campos
  let html = `<table>
    <thead>
      <tr>
        <th class="col-idx">#</th>
        <th>NOMBRE</th>
        <th>ENTRENADOR</th>
        <th>MÁX. GOLEADOR</th>
        <th>AÑO</th>
      </tr>
    </thead>
    <tbody>`;
 
  // 3. Recorremos el array con forEach e insertamos cada equipo
  datos.forEach(function(equipo, indice) {
    html += `<tr>
      <td class="col-idx">${indice + 1}</td>
      <td class="col-nombre">${equipo.nombre}</td>
      <td>${equipo.entrenador}</td>
      <td class="col-gol">${equipo.maximo_goleador_historico}</td>
      <td class="col-anio">${equipo.anio_creacion}</td>
    </tr>`;
  });
 
  html += "</tbody></table>";
 
  // 4. Insertamos el HTML generado en el DOM
  contenedor.innerHTML = html;
 
  // 5. Actualizamos el contador del hero
  document.getElementById("total").textContent = datos.length;
}
